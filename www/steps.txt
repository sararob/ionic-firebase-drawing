5. Add FB factory:

.factory('FBURL', function($firebaseAuth) {
  var usersRef = new Firebase('https://ionic-auth-demo.firebaseio.com/');
  var data = {
    auth: $firebaseAuth(usersRef),
    db: usersRef
  }
  return data;
})

6. Auth functions:

  $scope.login = function(authMethod) {
    FBURL.auth.$authWithOAuthRedirect(authMethod).then(function(authData) {
      console.log(authData);
    }).catch(function(error) {
      if (error.code === 'TRANSPORT_UNAVAILABLE') {
        FBURL.auth.$authWithOAuthPopup(authMethod).then(function(authData) {
          console.log(authData);
          $scope.modal.hide();
        });
      } else {
        console.log(error);
      }
    });
  };

  FBURL.auth.$onAuth(function(authData) {
    if (authData === null) {
      console.log('Not logged in yet');
    } else {
      console.log('Logged in as', authData.uid);
      ref.child('drawers').child(authData.uid).set(authData);
      // $scope.modal.hide();
      $ionicScrollDelegate.resize();

    }
    $scope.authData = authData; // This will display the user's name in our view
  });

8. Side menu to see who's online

    Save user data to Firebase with a Users factory:

        .factory('Users', function($firebaseObject) {
          var drawerRef = new Firebase('https://ionic-auth-demo.firebaseio.com/drawers');
          return $firebaseObject(drawerRef);
        })

    Define Users on the scope:

        $scope.users = Users;

    Edit $onAuth function to save the user data to Firebase

        $scope.users[authData.uid] = authData;
        $scope.users.$save();

    Add left sidebar to display user data:

    <!-- Left menu -->
    <ion-side-menu side="left">
      <ion-header-bar>

      </ion-content>
