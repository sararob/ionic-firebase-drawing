6. Add FB factory in factory.js:

.factory('FBURL', function($firebaseAuth) {

7. Add auth functions to AppCtrl:

.controller('AppCtrl', function($ionicScrollDelegate, $scope, FBURL, $ionicModal, $timeout, $ionicSideMenuDelegate, Users) {

  $scope.users = Users;

  $ionicModal.fromTemplateUrl('templates/modal.html', {
    scope: $scope
  }).then(function(modal) {
    $scope.modal = modal;
  });

  $scope.login = function(authMethod) {
    FBURL.auth.$authWithOAuthRedirect(authMethod).then(function(authData) {
      console.log(authData);
    }).catch(function(error) {
      if (error.code === 'TRANSPORT_UNAVAILABLE') {
        FBURL.auth.$authWithOAuthPopup(authMethod).then(function(authData) {
          console.log(authData);
          $scope.modal.hide();
        });
      } else {
        console.log(error);
      }
    });
  };

  FBURL.auth.$onAuth(function(authData) {
    if (authData === null) {
      console.log('Not logged in yet');
    } else {
      console.log('Logged in as', authData.uid);
      $ionicScrollDelegate.resize();

    }
    $scope.authData = authData; // This will display the user's name in our view
  });

})
